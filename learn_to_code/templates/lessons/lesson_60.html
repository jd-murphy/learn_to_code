nice_list = []
naughty_list = []
child = input("Add a child: (press enter twice to quit) ")
destination = input("Which list? ")
while child and destination:
  if destination.lower() == 'nice':
    nice_list.append(child)
  elif destination.lower() == 'naughty':
    naughty_list.append(child)
  else:
    print("Try adding the child again.")
  child = input("Add a child: (press enter twice to quit) ")
  destination = input("Which list? ")
print("nice_list")
print(nice_list)
print("naughty_list")
print(naughty_list)
